# SearchPanel 搜索面板

它的参数如下

  • onSearch: 搜索按钮按下时触发的回调，
  • searchSubmitTitle: 搜索按钮上的文字，
  • searchPlaceholder：搜索框的placeholder
interface SearchPanelPropsType {
    onSearch: ( key: string ) => any;
    searchSubmitTitle ?: string;
    searchPlaceholder ?: string;
};
